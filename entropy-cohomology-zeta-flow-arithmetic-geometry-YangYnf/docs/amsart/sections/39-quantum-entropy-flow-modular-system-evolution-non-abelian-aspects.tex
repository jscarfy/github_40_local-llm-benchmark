\subsection{Non-abelian symmetries in modular systems}

\begin{definition}[Non-abelian symmetries in modular systems]
In modular systems with non-abelian symmetries, the modular transformation operator satisfies the commutation relation:
\[
[\zeta^{[k]}, \zeta^{[\ell]}] = \mathsf{Q}^{k,\ell},
\]
where $\mathsf{Q}^{k,\ell}$ represents the quantum correction term modified by non-abelian symmetries.
\end{definition}

\subsection{Entropy flow evolution with non-abelian symmetries}

\begin{definition}[Entropy flow evolution with non-abelian symmetries]
The entropy flow operator in systems with non-abelian symmetries is defined as:
\[
\zeta^{[k]}(x) = x + \mathsf{Q}^{k}(x),
\]
where $\mathsf{Q}^{k}(x)$ reflects the quantum correction term influenced by non-abelian symmetries in the modular transformation process.
\end{definition}
